    <p>Stand: 3. Oktober 2025. Die Uhr tickt basierend auf Schätzungen des Bunds der Steuerzahler. Aktualisiere die Startwerte bei Bedarf.</p>

    <script>
        // Startwerte (anpassen, wenn neuere Daten verfügbar)
        let startDebt = 2636000000000; // 2.636 Billionen Euro (Stand 3. Okt 2025)
        let population = 83500000; // Ca. 83,5 Millionen
        let increasePerSecond = 5094; // Euro pro Sekunde

        // Funktion zum Formatieren von Zahlen mit Punkten und Euro-Zeichen
        function formatNumber(num) {
            return num.toLocaleString('de-DE') + ' €';
        }

        // Aktualisiere die Uhr
        function updateClock() {
            startDebt += increasePerSecond;
            const perCapita = Math.round(startDebt / population);

            document.getElementById('totalDebt').textContent = formatNumber(startDebt);
            document.getElementById('perCapita').textContent = formatNumber(perCapita);
        }

        // Initiale Anzeige
        updateClock();

        // Alle Sekunde aktualisieren
        setInterval(updateClock, 1000);
    </script>
